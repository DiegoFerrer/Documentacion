
<!--             Tipo de datos                    -->
<script>
	undefined 	// si una variable existe pero no tiene valor
	null 		// no existe
	NaN			// algo que no es un numero
	Boolean 	// valor true o false
	String		// cadena
	Number		// numero


// OPERADORES

	// Asignacion
		= 	// Le asigna un nuevo valor
		+= 	// a lo que ya tiene le suma el valor agregado
		-= 	// a lo que ya tiene le resta el valor agregado
		/= 	// a lo que ya tiene divide el valor agregado
		*= 	// a lo que ya tiene le multiplica el valor agregado
	// Comparacion
		== 	// Comparacion de valores
		!=	// Distinto de valor
		!== // Distinto de valor y tipo de dato
		=== // Comparacion de valor y tipo de dato
		>	// Mayor que
		<	// Menor que
		>=  // Mayor igual
		<=  // Menor igual
	// Unarios	
		typeof 	// devuelve el tipo de dato
		!		// negacion (devuelve un boolean)
	// Aritmeticos
		x-- 	// pre decremento (devuelve el decremento a la primera)
		--x 	// post decremento (devuelve el decremento a la segunda)
		x++		// pre incremento (devuelve el incremento a la primera)
		++x		// post incremento (devuelve el incremento a la segunda)
	// OTROS
		?	// expresion ? verdadero : falso
		||	// asignar uno u otro valor 

// Numeros

	toFixed(n) 				// especifica la cantidad de decimales
	parseInt('string',10) 	// Convierte un string a numero
	parseFloat('string')	// Convierte a numero pero conserva los decimales
	NaN 					// Cualquier operacion que no devuelva un numero 
</script>
